  4. Doing Simple Math :

    
#include<stdio.h>

int main(){
	int a,b,sm,sb,ml,dv;
	printf("Enter two numbers to do simple math:");
	scanf("%d%d", &a, &b);
	sm=a+b;
	sb=a-b;
	ml=a*b;
	dv=a/b;
	printf("addition=%d\nsubtraction=%d\nmultiplication=%d\ndivision=%d\n",sm,sb,ml,dv);

}

  5. Swapping Using 3rd Variable :

    
#include<stdio.h>

int main(){
	int a,b,temp;
	printf("Enter two numbers for swapping");
	printf("\nA= ");
	scanf("%d", &a);
	printf("\nB= ");
	scanf("%d", &b);
	temp=a;
	a=b;
	b=temp;
	printf("\nAfter swapping A=%d and B=%d\n",a,b);
	return 0;
}

  7. Temperature Fahrenheit to Celsius Convertion

    
#include <stdio.h>

int main() {
    float fahrenheit, celsius;

    // Input temperature in Fahrenheit
    printf("Enter temperature in Fahrenheit: ");
    scanf("%f", &fahrenheit);

    // Convert Fahrenheit to Celsius
    celsius = (5.0 / 9.0) * (fahrenheit - 32.0);

    // Output temperature in Celsius
    printf("Temperature in Celsius: %.2f\n", celsius);

    return 0;
}

  8. Calculate No. of Years and Weeks From Total No. of Days :

    
#include<stdio.h>

int main(){
	int n,y,w,d;
	printf("Enter number of days: ");
	scanf("%d", &n);
	y=n/365;
	w=(n%365)/7;
	d=(n%365)%7;
	printf("%d = %d years, %d weeks, %d days\n",n,y,w,d);
}

  9. Calculate Gross Salary :

    
#include<stdio.h>

int main(){
	float bp,da,hra,gs,pf;
	printf("Enter Basic pay: ");
	scanf("%f", &bp);

	hra=0.2*bp;
	da=0.4*bp;
	pf=0.12*bp;
	gs=bp+da+hra+pf;

	printf("Basic Pay = %.2f\n",bp);
	printf("Dearness Allowance = %.2f\n",da);
	printf("House Rent Allowance = %.2f\n", hra);
	printf("Provident Fund = %.2f\n", pf);
	printf("Gross Salary = %.2f\n", gs);

	return 0;
}
